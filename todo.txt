/* per la modifica in ricetta.js*/
export async function salvaModifiche(id) {
    const nuovoTitolo = document.getElementById('edit-titolo').value;
    const nuovaEsecuzione = document.getElementById('edit-esec').value;

    const { error } = await _supabase
        .from('ricette')
        .update({ 
            Titolo: nuovoTitolo, 
            Esecuzione: nuovaEsecuzione 
        })
        .eq('pkRicetta', id);

    if (error) {
        alert("Errore durante il salvataggio: " + error.message);
    } else {
        alert("Ricetta aggiornata con successo!");
        showRicetta(id); // Ricarica la pagina in modalitÃ  visualizzazione
    }
}

Passaggio C: Policy di Update su Supabase (Fondamentale!)
Prima di poter salvare, devi andare su Supabase:

Vai in Authentication -> Policies.

Tabella ricette -> New Policy.

Scegli "Enable update for all users" (o crea una custom con operazione UPDATE).

In Check Expression, scrivi true.

Salva.
